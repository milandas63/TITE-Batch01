=	for loop
	Syntax:
		for([initializer]; [condition]; [increment/decrement]) [{}]

	Example:
    (1)
		for(;;)
            System.out.println("Hello World!");
            System.out.println("I am Java");

    (2)
        for(int i=0; i<5; i++) {
            System.out.println("Hello World!");
        }
        
    (3)
        int num = 25;
        for(;num>20;) {
            System.out.println("Hello World!");
            num = num-1;
        }

    (4)
        for(int i=5, j=0, k=7; i>0 && k>4; i--, j++, k--) {
            System.out.println("Hello World!");
        }

    (5)
        int num = 25;
        for(;;) {
            System.out.println("Hello World!");
            num = num-1;
            if(num<=20) {
                break;
            }
        }

                ok = false;
            }
        }

    public class AddRecurrsive {
        int multiply(n, o) {
            int prod = 0;
            for(int i=0; i<o; i++) {
                prod = prod+n;
            }
            return prod;
        }
    }

    class Multiply {
        public static void main(String[] args) {
            AddRecurrsive ar = new AddRecurrsive();
            int result = ar.multiply(3,4);
        }
    }

    class Multiply extends AddRecurrsive {
        public int compute() {
            int result = multiply(3,4);
        }
    }
